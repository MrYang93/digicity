### 面向对象小案例


```
class Block {
  constructor(width, height) {
    this.width = width;
    this.height = height;
  }
  area() {
    console.log(this.width * this.height);
  }
  addWidthHeight() {
    console.log(this.width + this.height);
  }
}

let block1 = new Block(2,3);
let block2 = new Block(4,7);
block1.area();
block1.addWidthHeight();
block2.area();
block2.addWidthHeight();
```
